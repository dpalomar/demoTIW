22  /**
23   * Servlet de ejemplo que muestra distintos aspectos dentro de los ambitos request y session. 
24   * -Como usar el metodo init para inicializar datos
25   * -Como recoger datos desde un formulario por post
26   * -Como no permitir acceso por get redirigiendo a la pagina de login.jsp
27   * -Uso del objeto RequestDispatcher y forward
28   * -Introduccion de atributos en el objeto request
29   * -Uso de objeto sesion para mantener al usuario autenticado
30   * -Control de flujo y logica de negicio de un controlador.
31   * 
32   */
33  @WebServlet(value="/login",loadOnStartup=1)
34  public class LoginServlet extends HttpServlet {
35  
36  	private static final String ENTRADA_JSP = "/listado.jsp";
37  	private static final String LOGIN_JSP = "/login.jsp";
38  	private static final long serialVersionUID = 1L;
39  	private es.uc3m.tiw.model.Usuario usuario;
40  	private List<Usuario> usuarios;
41  	@PersistenceContext(unitName="demo-model")
42  	private EntityManager em;
43  	@Resource
44  	private UserTransaction ut;
45  	private PersonaDao dao;
46  	@Override
47  	public void init() throws ServletException {
48  	
49  		dao = new PersonaDao(em, ut);
50  		
51  		
52  	}
53         
54  
55  
56  	/**
57  	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
58  	 */
59  	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
60  		String salir = request.getParameter("accion");
61  		if (salir != null && !salir.equals("")) {
62  			request.getSession().invalidate();
63  		}
64  		this.getServletContext().getRequestDispatcher(LOGIN_JSP).forward(request, response);
65  	}
66  
67  	/**
68  	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
69  	 */
70  	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
71  		
72  		String user = request.getParameter("usuario");
73  		String password = request.getParameter("password");
74  		String mensaje ="";
75  		String pagina = "";
76  		pagina = LOGIN_JSP;
77  		HttpSession sesion = request.getSession(true);
78  		Usuario u = comprobarUsuario(user, password);
79  		if (u != null){
80  			
81  			pagina = ENTRADA_JSP;
82  			
83  			try {
84  			
85  				usuarios = dao.findAll();
86  				
87  			} catch (InstantiationException | IllegalAccessException
88  					| ClassNotFoundException | SQLException e) {
89  				// TODO Auto-generated catch block
90  				e.printStackTrace();
91  			}
92  			
93  			request.setAttribute("usuarios", usuarios);
94  			sesion.setAttribute("usuario", u);
95  			sesion.setAttribute("acceso", "ok");
96  			
97  		}else{
98  			
99  			mensaje = "Usuario o password incorrectos";
100 			request.setAttribute("mensaje", mensaje);
101 		}
102 		
103 			this.getServletContext().getRequestDispatcher(pagina).forward(request, response);
104 			
105 		
106 	}
107 
108 	private Usuario  comprobarUsuario(String user, String password) {
109 		Usuario usuario = dao.findByNickAndPassword(user,password);
110 		return usuario;
111 	}

113 }
